def bubble_sort_optimized(arr):
    n = len(arr)
    
    for i in range(n - 1):
        swapped = False  # –§–ª–∞–≥ –æ–±–º–µ–Ω–æ–≤
        
        for j in range(n - i - 1):
            if arr[j] > arr[j + 1]:
                arr[j], arr[j + 1] = arr[j + 1], arr[j]
                swapped = True
        
        # –ï—Å–ª–∏ –æ–±–º–µ–Ω–æ–≤ –Ω–µ –±—ã–ª–æ - –º–∞—Å—Å–∏–≤ –æ—Ç—Å–æ—Ä—Ç–∏—Ä–æ–≤–∞–Ω
        if not swapped:
            break
    
    return arr
